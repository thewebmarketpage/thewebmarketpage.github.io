async function fetchData() { const response = await fetch('data.json'); return response.json(); } // Global variable to store all data let allProducts = []; // Number of items to load initially and each time more are loaded const itemsPerLoad = 9; // Track how many items are currently displayed let currentlyLoaded = 0; function lazyLoadCards(data, searchQuery = '') { const cardsContainer = document.getElementById('cardsContainer'); // Filter data based on search query const filteredData = data.filter(item => item.title.toLowerCase().includes(searchQuery.toLowerCase()) || item.category.toLowerCase().includes(searchQuery.toLowerCase()) || item.description.toLowerCase().includes(searchQuery.toLowerCase()) ); // Clear container on new search if (searchQuery !== '' || currentlyLoaded === 0) { cardsContainer.innerHTML = ''; currentlyLoaded = 0; } // Calculate end index for current batch const endIndex = Math.min(currentlyLoaded + itemsPerLoad, filteredData.length); // Get the current batch of items to display const itemsToShow = filteredData.slice(currentlyLoaded, endIndex); // Create and append cards for each item in the current batch itemsToShow.forEach(item => { const card = document.createElement('div'); card.className = "bg-gray-800 rounded-lg shadow-lg border-b-4 border-blue-500 transform transition duration-300 hover:scale-105 hover:shadow-xl"; card.innerHTML = `
${item.category}

${item.title}

${item.description.substring(0, 100)}...

View Product
`; cardsContainer.appendChild(card); }); // Update the count of loaded items currentlyLoaded = endIndex; // Show or hide the load more button based on whether there are more items updateLoadMoreButton(currentlyLoaded < filteredData.length); } function updateLoadMoreButton(show) { let loadMoreButton = document.getElementById('loadMoreButton'); if (!loadMoreButton && show) { // Create the button if it doesn't exist and needs to be shown loadMoreButton = document.createElement('button'); loadMoreButton.id = 'loadMoreButton'; loadMoreButton.className = 'mt-8 mx-auto block bg-blue-500 text-white py-2 px-6 rounded-full font-semibold hover:bg-blue-600 transition duration-300'; loadMoreButton.textContent = 'Load More Products'; loadMoreButton.addEventListener('click', () => lazyLoadCards(allProducts, document.getElementById('searchBox').value)); const container = document.querySelector('.max-w-7xl'); container.appendChild(loadMoreButton); } else if (loadMoreButton) { // Update existing button visibility loadMoreButton.style.display = show ? 'block' : 'none'; } } // Intersection Observer to detect when user scrolls near the bottom function setupIntersectionObserver() { const options = { root: null, rootMargin: '0px', threshold: 0.1 }; const observer = new IntersectionObserver((entries) => { entries.forEach(entry => { if (entry.isIntersecting) { // Load more cards when the sentinel is visible lazyLoadCards(allProducts, document.getElementById('searchBox').value); } }); }, options); // Create and observe a sentinel element at the bottom of the cards container const sentinel = document.createElement('div'); sentinel.id = 'sentinel'; sentinel.style.height = '10px'; document.getElementById('cardsContainer').after(sentinel); observer.observe(sentinel); } document.addEventListener('DOMContentLoaded', async () => { try { // Add loading indicator const cardsContainer = document.getElementById('cardsContainer'); cardsContainer.innerHTML = '

Loading products...

'; // Fetch data allProducts = await fetchData(); // Initial load of cards lazyLoadCards(allProducts); // Setup search functionality document.getElementById('searchBox').addEventListener('input', (e) => { lazyLoadCards(allProducts, e.target.value); }); // Setup intersection observer for automatic loading setupIntersectionObserver(); } catch (error) { console.error('Error loading products:', error); document.getElementById('cardsContainer').innerHTML = '
Error loading products. Please try again later.
'; } });
